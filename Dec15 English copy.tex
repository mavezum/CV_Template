\newcommand{\score}[2]{%
  \begin{tikzpicture}
    \foreach \x in {1,...,#2} {%
	    \ifnum \x > #1
	    \draw[fill=white] (\x*\myscale,-1) circle (0.1cm);
	    \else
	    \draw[fill=myblue] (\x*\myscale,-1) circle (0.1cm);
	    \fi
  }
  \end{tikzpicture}%
}
